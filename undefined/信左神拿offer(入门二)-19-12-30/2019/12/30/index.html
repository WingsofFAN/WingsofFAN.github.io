递归行为

  • 定义：程序调用自身的编程技巧称为递归（ recursion）。

  • 条件：

    • 子问题须与原始问题为同样的事，且更为简单；
    • 不能无限制地调用本身，须有个出口，化简为非递归状况处理

  • master 公式:

    N母问题的数据量；子问题的数据量N/b，调用次数为a；O(NN)除去子问题后的其他流程的时间复杂度。子问题规模一样的递归才能使用master公式估计时间复杂度。

求数组中点位置：

//普通写法
int mid =  (L+R)/2
//由于L和R可能会过大，所以相加可能有溢出的危险

//防止溢出的写法
int mid = L +(R-L)>>1;
//加会溢出，但是换成减就不会
//除二换成右移一位，可以加快运算

归并排序

  • 流程：

    • 从中点分开先让左侧有序，再让右侧有序
    • 若无序则递归
    • 合并有序两个序列的

  • 代码

void merge(vector<int>& arr,int L,int M,int R)
{
    vector<int> help(R-L+1);
    int i = 0;
    int p1 = L;
    int p2 = M+1;
    while(p1<=M && p2 <= R)
    {
        help[i++] = arr[p1] <= arr[p2] : arr[p1++] ? arr[p2++];
    }
    while(p1 <= M)
    {
        help[i++] = arr[p1++];
    }
    while(p2 <= R)
    {
        help[i++] = arr[p2++];
    }
    for (i = 0;i < help.length;i++)
    {
        arr[L+i] = help[i];
    }
    help.release();
}


void mergeSort(vector<int> arr,int L,int R)
{
    if(L == R)
    {
        return;            //递归出口
    }
    int mid = L+(R-L)>>1;
    mergeSort(arr,L,mid);
    mergeSort(arr,mid+1,R);
    merge(arr,L,mid,R);
}

  • 优势：高效利用了比较行为，简单排序算法每排序一个数都是要进行单独一轮的比较，而并没有利用到上一个数比较时的结果。而归并在每一次比较后都被整体有序的部分保存了下来，提供给merge时使用。

  • ※小和问题：

    • 在一个数组中，每一个数左边比当前数小的数累加起来，叫做这个数组

      的小和。求一个数组 的小和。

    • 例子:[1,3,4,2,5] 1左边比1小的数，没有; 3左边比3小的数，1; 4左

      边比4小的数，1、3; 2左边比2小的数，1; 5左边比5小的数，1、3、4、

      2; 所以小和为1+1+3+1+1+3+4+2=16

    ```c++
    int merge(vector& arr,int L,int M,int R)
    {

    vector<int> help(R-L+1);
int i = 0;
int p1 = L;
int p2 = M+1;
int res = 0;
while(p1<=M && p2 <= R)
{
    res += arr[p1] < arr[p2] ? (r - p2 + 1)*arr[p1]:0;   //求小和
    help[i++] = arr[p1] <= arr[p2] : arr[p1++] ? arr[p2++];
}
while(p1 <= M)
{
    help[i++] = arr[p1++];
}
while(p2 <= R)
{
    help[i++] = arr[p2++];
}
for (i = 0;i < help.length;i++)
{
    arr[L+i] = help[i];
}
help.release();

    }

int process(vector arr,int L,int R)
{
if(L == R)
{
return 0; //递归出口
}
int mid = L+(R-L)>>1;
return process(arr,L,mid) 

      +  process(arr,mid+1,R)
      +  merge(arr,L,mid,R);

}

int smallSum(vector& arr)
{
if(arr == NULL || arr.size() < 2)
{
return 0;
}
return process(arr,0,arr.size()-1);
}

  构成一个逆序对，请打印所有逆序对。

  ```c++
  int merge(vector<int>& arr,int L,int M,int R)
  {
      vector<int> help(R-L+1);
      int i = 0;
      int p1 = L;
      int p2 = M+1;
      int res = 0;
      while(p1<=M && p2 <= R)
      {
          res += arr[p1] > arr[p2] ? (r - p2 + 1):0;               //求逆序次数
          help[i++] = arr[p1] >= arr[p2] : arr[p1++] ? arr[p2++]; //逆排序
      }
      while(p1 <= M)

  • 荷兰国旗：

    • 给定一个数组arr，和一个数num，请把小于等于num的数放在数组的左边，大于num的

      数放在数组的右边。要求额外空间复杂度O(1)，时间复杂度O(N)

    • 流程

      [i] <=num,[i]和<=区的下一个数交换，<=区右扩，i++

      [i]>num,i++

    • 给定一个数组arr，和一个数num，请把小于num的数放在数组的 左边，等于num的数放

      在数组的中间，大于num的数放在数组的 右边。要求额外空间复杂度O(1)，时间复杂度

      O(N)

    • 流程

      [i]<num,[i]和<区下一个交换，<区右扩i++

      [i]==num,i++

      [i]>num,[i]和>区前一个交换，>区左扩，i不变

  • 快速排序：

    • 1.0版：递归以每一个范围上的最后一个做划分，并将其交换到<=区最后一个，每次即可确定一个数的位置
    • 2.0版: 递归以每一个范围上的最后一个做划分，并将其交换到=区，再递归<区和>区，每次即可确定一批数的位置

    前两个版面对最差的情况时，时间复杂度为O(nn)。划分值打的偏则，情况越差；划分值打在中点，情况越好O(nlogn)。

    • 3.0版:划分值在区域中随机选择，则避免了确定的最差情况，带入概率求期望后时间复杂度为O(nlogn)

    ```c++
    void swap(vector arr,int a,int b)
    {

    int t = arr[a];
arr[a] = arr[b];
arr[b] = t ;

    }

    vector partition(vector& arr,int L,int R)
    {

    int less = L-1; //
int more = R;
while(L < more)
{
    if(arr[L] < arr[R])
    {
        swap(arr,++less,L++);
    }
    else if (arr[L] > arr[R] )
    {
        swap(arr,--more,L);
    }
    else
    {
        L++;
    }
}
swap(arr,more,R);
return vector<int>{less+1,more};

    }

void quickSort(vector arr,int L,int R)
{
if(L < R)
{
swap(arr,L+rand()%(R-L+1),R);
vector p= partition(arr,L,R); //返回标志位
quickSort(arr,L,p[0] - 1);
quickSort(arr,p[1]+1,R);
}
}
```

而外空间复杂度：O(logN) 同样是依赖概率来求的，最差为O(N)，最好为O(logN)。
